### 백준 25631
* 마트료시카 합치기

```
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  
  map<int, int> _map;
  
  int n = 0;
  cin >> n;
  
  while(n--) {
    int v = 0;
    cin >> v;
    
    _map[v]++;
  }
  
  int _max = 0;
  for (auto m : _map) {
    _max = max(_max, m.second);
  }
  
  cout << _max;
  
  return 0;
}
```
